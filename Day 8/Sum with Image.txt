import 'package:flutter/material.dart';
import 'package:fluttertoast/fluttertoast.dart';

void main() {
  runApp(Home());
}

class Home extends StatelessWidget {
  const Home({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: "App",
      debugShowCheckedModeBanner: false,
      theme: ThemeData(primarySwatch: Colors.indigo),
      home: HomePage(),
    );
  }
}

class HomePage extends StatefulWidget {
  const HomePage({Key? key}) : super(key: key);
  @override
  State<HomePage> createState() => _HomePageState();
}

class _HomePageState extends State<HomePage> {

  TextEditingController txt1 = new TextEditingController();
  TextEditingController txt2 = new TextEditingController();
  var store = "";

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text("CALCULATOR", style: TextStyle(color: Colors.white)),

      ),
      body: Container(
        padding: EdgeInsets.all(32),
        child: ListView(
          children: [

            SizedBox(height: 20.0,),

            Image.asset('assets/images/abc.jpg'),

            SizedBox(height: 40.0,),

            // Text("HELLO", style: TextStyle(fontFamily: "Poppins", fontSize: 40),),
            // Text("HELLO", style: TextStyle(fontSize: 40),),

            TextField(
              textAlign: TextAlign.center,
              keyboardType: TextInputType.number,
              decoration: InputDecoration(
                border: OutlineInputBorder(
                  borderRadius: BorderRadius.circular(10.0),
                ),
                labelText: "Enter Value",
              ),
              controller: txt1,
            ),

            SizedBox(
              height: 15.0,
            ),

            TextField(
              textAlign: TextAlign.center,
              keyboardType: TextInputType.number,
              decoration: InputDecoration(
                border: OutlineInputBorder(
                  borderRadius: BorderRadius.circular(10.0),
                ),
                labelText: "Enter Value",
              ),
              controller: txt2,
            ),
            
            Row(
              children: [

                Container(
                  child: Expanded(
                    child: ElevatedButton(
                      onPressed: () {
                        calculation();
                      },
                      child: Text('ADDITION'),
                    ),
                  ),
                ),
                
              ],
            ),

            SizedBox(height: 20.0,),

            Center(child: Text("${store}", style: TextStyle(fontSize: 20.0, fontWeight: FontWeight.bold)),),
            

          ],
        ),
      ),
    );
  }

  void calculation() {
    var text_1 = txt1.text;
    var text_2 = txt2.text;
    var c = int.parse(text_1) + int.parse(text_2);
    toast("check you sum answer");
    setState(() {
      store = c.toString();
    });
  }

  void toast(value) {
    Fluttertoast.showToast(
        msg: value,
        toastLength: Toast.LENGTH_SHORT,
        gravity: ToastGravity.BOTTOM,
        timeInSecForIosWeb: 1,
        backgroundColor: Colors.indigo,
        textColor: Colors.white,
        fontSize: 16.0,
    );
  }
}
